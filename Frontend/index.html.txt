<script>
let contract;

const address = "0x5FbDB2315678afecb367f032d93F642f64180aa3";

const abi = [
  "function donate() public payable",
  "function getDonations() public view returns (tuple(address donor,uint amount)[])"
];

async function connectWallet() {
  if (!window.ethereum) {
    alert("Install MetaMask");
    return;
  }

  await window.ethereum.request({ method: "eth_requestAccounts" });

  const provider = new ethers.BrowserProvider(window.ethereum);
  const signer = await provider.getSigner();

  contract = new ethers.Contract(address, abi, signer);

  alert("Wallet Connected!");
}

async function donate() {
  const amt = document.getElementById("amount").value;

  const tx = await contract.donate({
    value: ethers.parseEther(amt)
  });

  await tx.wait();

  alert("Donation Successful!");
}

async function getDonations() {
  const data = await contract.getDonations();
  const list = document.getElementById("list");
  const totalDiv = document.getElementById("total");

  list.innerHTML = "";

  let total = 0;

  data.forEach(d => {
    const amount = parseFloat(ethers.formatEther(d.amount));
    total += amount;

    const li = document.createElement("li");
    li.innerText = d.donor + " donated " + amount + " ETH";
    list.appendChild(li);
  });

  totalDiv.innerText = "Total Donations: " + total + " ETH";
}
</script>
